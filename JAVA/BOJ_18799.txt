import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Deque;
import java.util.Iterator;
import java.util.LinkedList;

public class Main {

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		Iterator<Character> itr;
		String writing = br.readLine();
		String wTmp = "";
		int count = 0;
		int i = 0;
		Deque<Character> stack = new LinkedList<Character>();
		
		while(!writing.equals(wTmp)) {
			char c = writing.charAt(i);
			
			if(stack.size() != 0 && stack.peekFirst() == c) {
				itr = stack.iterator();
				int ii = i;
				int p = 0;
				boolean flag = true;
				while(itr.hasNext()) {
					
					if(itr.next() == writing.charAt(ii)) {
						p++;
					}else {
						p = 0;
						flag = false;
						break;
					}
					ii++;
				}
				
				if(flag) {
					itr = stack.iterator();
					while(itr.hasNext()) wTmp += itr.next().toString();
				}
				i += p-1;
			}else {
				stack.add(c);
			}
			i++;
			count++;
			if(i > writing.length()-1) {
				count++;
				itr = stack.iterator();
				while(itr.hasNext()) wTmp += itr.next().toString();
				break;
			}
		}//end while
		
		System.out.println(count);
	}

}
